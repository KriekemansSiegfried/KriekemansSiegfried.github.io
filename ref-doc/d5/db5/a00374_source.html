51 CSV::CSV(const boost::filesystem::path& path, initializer_list<string> optLabels) : columnCount(0)
52 {
53  try {
54  boost::filesystem::path full_path = check(path);
55  boost::filesystem::ifstream file;
56  file.open(full_path.string());
57  if (!file.is_open()) {
58  throw runtime_error("Error opening csv file: " + full_path.string());
59  }
60  string line;
61 
62  // header
63  getline(file, line);
64  line = Trim(line);
65  // Split is bad! There is no option to escape ",".
66  vector<string> headerLabels = Split(line, ",");
67  for (const string& label : headerLabels) {
68  labels.push_back(Trim(label, "\""));
69  }
70  columnCount = labels.size();
71 
72  // body
73  while (getline(file, line)) {
74  line = Trim(line);
75  if (!line.empty()) {
76  // Split is bad! There is no option to escape ",".
77  vector<string> values = Split(line, ",");
78  AddRow(values);
79  }
80  }
81  } catch (runtime_error& error) {
82  if (optLabels.size() == 0) {
83  throw error;
84  } else {
85  labels = optLabels;
86  columnCount = labels.size();
87  }
88  }
