59 void SimRunner::Run(unsigned int numSteps)
60 {
61  // Saveguard against repeatedly firing AtStart, so bypass if numSteps == 0.
62  if (numSteps != 0U) {
63  // Prelims.
64  m_clock.Start();
65  const auto numDays = m_config_pt.get<unsigned int>("run.num_days");
66 
67  // We are AtStart: no steps have taken yet, so signal AtStart.
68  if (m_sim->GetCalendar()->GetSimulationDay() == 0) {
69  Notify(Id::AtStart);
70  }
71 
72  // Take numSteps but do not go beyond numDays.
73  for (unsigned int i = 0; i < numSteps; i++) {
74  // This is not the last step: execute and signal Stepped.
75  if (m_sim->GetCalendar()->GetSimulationDay() < numDays - 1) {
76  m_sim->TimeStep();
77  Notify(Id::Stepped);
78  // This is the last step so execute and afterwards signal Stepped and Finished
79  } else if (m_sim->GetCalendar()->GetSimulationDay() == numDays - 1) {
80  m_sim->TimeStep();
81  Notify(Id::Stepped);
82  Notify(Id::Finished);
83  break;
84  // We are apparently already at the end of the numDays so nothing to do or signal.
85  } else {
86  break;
87  }
88  }
89 
90  m_clock.Stop();
91  }
92 }
