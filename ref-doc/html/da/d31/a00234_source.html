226 qreal Stats::FindMedian(const std::vector<qreal>& list, int begin, int end)
227 {
228  // source: https://doc.qt.io/qt-5.10/qtcharts-boxplotchart-boxdatareader-cpp.html
229 
230  const int count = end - begin;

232  return list.at(static_cast<unsigned long>(double(count) / 2.0 + begin));
