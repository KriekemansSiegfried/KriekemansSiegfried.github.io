444 QPointF Window::TranslateToOriginal(const QPointF& pos) const
445 {
446  const double hLengthPicture = image.width();
447  const double vLengthPicture = image.height();
448  const double hLengthWindow = scrollAreaImage->width() / scaleFactor;
449  const double vLengthWindow = scrollAreaImage->height() / scaleFactor;
450 
451  const double vMargin = vLengthPicture - vLengthWindow;
452  double vStart = 0;
453  if (vValue > 0) {
454  const double vPercentage = vValue / vScrollMax;
455  vStart = vMargin * vPercentage;
456  }
457 
458  const double hMargin = hLengthPicture - hLengthWindow;
459  double hStart = 0;
460  if (hValue > 0) {
461  const double hPercentage = hValue / hScrollMax;
462  hStart = hMargin * hPercentage;
463  }
464 
465  return QPointF(pos.x() / scaleFactor + hStart, pos.y() / scaleFactor + vStart);
466 }

468 stride::Point2D Window::ConvertLatLongToPixel(double latitude, double longitude) const
469 {
470  const double mapWidth = image.width();
471  const double mapHeight = image.height();
472 
473  // https://stackoverflow.com/questions/38748832/convert-longitude-and-latitude-coordinates-to-image-of-a-map-pixels-x-and-y-coor
474  const double mapLongitudeStart = LEFTUPPER_LONGITUDE, mapLatitudeStart = LEFTUPPER_LATITUDE;
475  const double mapLongitude = RIGHTUNDER_LONGITUDE - mapLongitudeStart,
476  mapLatitude = mapLatitudeStart - RIGHTUNDER_LATITUDE;
477 
478  longitude -= mapLongitudeStart;
479  latitude = mapLatitudeStart - latitude;
480  const double x = mapWidth * (longitude / mapLongitude);
481  const double y = mapHeight * (latitude / mapLatitude);
482 
483  return stride::Point2D(x, y);
484 }
