95 void CliController::Control()
96 {
97  // -----------------------------------------------------------------------------------------
98  // Build population, instantiate SimRunner & register viewers & run.
99  // -----------------------------------------------------------------------------------------
100  // Check if run has multiple populations
101  if (m_config_pt.get<unsigned int>("run.amount_of_populations", 1) > 1) {
102  auto multipop = std::make_shared<stride::MultiPopulation>(m_config_pt);
103  auto pop = multipop->Get();
104 
105  if (m_config_pt.get<bool>("run.output_map") || m_config_pt.get<bool>("run.output_png")) {
106  auto runner = make_shared<GuiRunner>(m_config_pt, multipop);
107  RegisterViewers(runner);
108  RegisterQtViewers(runner);
109  runner->Run();
110  if (m_config_pt.get<bool>("run.output_map")) {
111  // Still allow the user to use the gui (for example, start a new simulation)
112  app.exec();
113  }
114  } else {
115  auto runner = make_shared<SimRunner>(m_config_pt, multipop);
116  RegisterViewers(runner);
117  runner->Run();
118  }
119  } else {
120  auto pop = Population::CreateEmpty(
121  m_config_pt); // create empty population (initializes logger but does not create pop)
122  auto grid = std::make_shared<stride::GeoGrid>(pop); // create grid with empty population
123  pop = Population::Create(m_config_pt, grid);
124  if (m_config_pt.get<bool>("run.output_map") || m_config_pt.get<bool>("run.output_png")) {
125  auto runner = make_shared<GuiRunner>(m_config_pt, grid);
126  RegisterViewers(runner);
127  RegisterQtViewers(runner);
128  runner->Run();
129  if (m_config_pt.get<bool>("run.output_map")) {
130  // Still allow the user to use the gui (for example, start a new simulation)
131  app.exec();
132  }
133  } else {
134  auto runner = make_shared<SimRunner>(m_config_pt, pop);
135  RegisterViewers(runner);
136  runner->Run();
137  }
138  }
139 
140  m_stride_logger->info("CliController shutting down.");
141  spdlog::drop_all();
142 }

161 void CliController::RegisterViewers(shared_ptr<SimRunner> runner)
162 {
163  // Command line viewer
164  m_stride_logger->info("Registering CliViewer");
165  const auto cli_v = make_shared<viewers::CliViewer>(runner, m_stride_logger);
166  runner->Register(cli_v, bind(&viewers::CliViewer::Update, cli_v, placeholders::_1));
167 
168  // Adopted viewer
169  if (m_config_pt.get<bool>("run.output_adopted", false)) {
170  m_stride_logger->info("registering AdoptedViewer,");
171  const auto v = make_shared<viewers::AdoptedViewer>(runner, m_output_prefix);
172  runner->Register(v, bind(&viewers::AdoptedViewer::Update, v, placeholders::_1));
173  }
174 
175  // Infection counts viewer
176  if (m_config_pt.get<bool>("run.output_cases", false)) {
177  m_stride_logger->info("Registering InfectedViewer");
178  const auto v = make_shared<viewers::InfectedViewer>(runner, m_output_prefix);
179  runner->Register(v, bind(&viewers::InfectedViewer::Update, v, placeholders::_1));
180  }
181 
182  // Persons viewer
183  if (m_config_pt.get<bool>("run.output_persons", false)) {
184  m_stride_logger->info("registering PersonsViewer.");
