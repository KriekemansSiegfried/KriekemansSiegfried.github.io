141 
142  // population->reserve(real_pop_size); // Avoid reallocation: we need valid pointers

144  populateCaches();
145  unsigned personId = 0;
146  const unsigned firstHouseholdId =
147  geogrid->GetContactPoolIdRange(GridPoolType::Id::SecondaryCommunity, geogrid->NumLocations() - 1).second -
148  1;
149  const auto populationSizes = geogrid->GetLocationPopulationsize();
150  const trng::fast_discrete_dist locationDist{populationSizes.begin(), populationSizes.end()};
151  //#pragma omp parallel // num_threads(1)
152  // {
153  const auto locationIndexGenerator =
154  rn_manager->GetGenerator(locationDist, static_cast<size_t>(omp_get_thread_num()));
155  const auto studentCommuteGenerator =
156  rn_manager->GetGenerator(student_commute_dist, static_cast<size_t>(omp_get_thread_num()));
157  const auto activeCommuteGenerator =
158  rn_manager->GetGenerator(active_commute_dist, static_cast<size_t>(omp_get_thread_num()));
159 
160  auto households = geogrid->GetHouseHolds();
161  //#pragma omp for
162 
163  for (unsigned householdIndex = 0; householdIndex < households.size(); ++householdIndex) {
164  const auto locIndex = static_cast<LocIndex>(locationIndexGenerator());
165  unsigned primary_community_id;
166  primary_community_id = pickPool(GridPoolType::Id::PrimaryCommunity, locIndex);
167  for (unsigned i = 0; i < households[householdIndex].occupations.size(); ++i) {
168  unsigned school_id = 0;
169  unsigned work_id = 0;

171  secondary_community_id = pickPool(GridPoolType::Id::SecondaryCommunity, locIndex);
172  switch (households[householdIndex].occupations[i]) {
173  case Occupation::Nonactive: {
174  break;
175  }
176  case Occupation::SchoolStudent: {
177  school_id = pickPool(GridPoolType::Id::School, locIndex);

179  }

186  break;
187  }

190  auto picked = pickPoolCommuting(GridPoolType::Id::Workplace, locIndex);
191  work_id = picked.second;
192  auto starting_id =
193  geogrid->GetContactPoolIdRange(GridPoolType::Id::Workplace, picked.first)
194  .first;
195  auto id = work_id - starting_id;
196  workplaceSizes[picked.first][id] += 1;
197 
198  } else {
199  work_id = pickPool(GridPoolType::Id::Workplace, locIndex);
200  auto starting_id =
201  geogrid->GetContactPoolIdRange(GridPoolType::Id::Workplace, locIndex).first;
202  auto id = work_id - starting_id;
203  workplaceSizes[locIndex][id] += 1;
204  }
205 
206  break;
207  }
208  }
209  //#pragma omp critical(addPerson)
210  {
211  geogrid->AddPerson(locIndex, personId, households[householdIndex].ages[i],
212  householdIndex + firstHouseholdId, school_id, work_id,
213  primary_community_id, secondary_community_id);

223 void stride::GeoGridPopulator::populateCaches()
224 {
225  for (const auto poolType : GridPoolType::Types) {
226  contactPoolDistributions[poolType].resize(geogrid->NumLocations());
227  // For each location:
228  for (LocIndex locIndex = 0; locIndex < geogrid->NumLocations(); ++locIndex) {
229  // Search for contact pools in a radius.
230  double minRadius = 0.0;
231  double maxRadius = 10.0;
232  // While no contact pools found in radius:
233  while (contactPoolDistributions[poolType][locIndex].empty()) {
234  const auto locations = geogrid->GetLocationsWithinRange(locIndex, minRadius, maxRadius);
235  for (const auto& loc : locations) {
236  const auto idRange = geogrid->GetContactPoolIdRange(poolType, loc);
237  for (unsigned id = idRange.first; id < idRange.second; ++id) {
238  contactPoolDistributions[poolType][locIndex].emplace_back(id);
239  }
240  }
241  minRadius = maxRadius;
242  maxRadius *= 2;
243  }
244  }
245  }

256 std::pair<unsigned, unsigned> stride::GeoGridPopulator::pickPoolCommuting(const stride::GridPoolType::Id poolType,
257  const stride::LocIndex locIndex)
258 {
259  unsigned location = 0;
260  // Get the location where the commuter will commute to.
261  switch (poolType) {
262  case GridPoolType::Id::Academy: {
263  const auto cities = geogrid->GetLargestLocations();
264  std::vector<unsigned> commutingDist;
265  for (const auto cityLoc : cities) {
266  if (geogrid->GetNumberOfAcademies(cityLoc) > 0) {
267  commutingDist.emplace_back(geogrid->GetCommuteValue(locIndex, cityLoc));
268  }
269  }
270 
271  trng::fast_discrete_dist locationDist{commutingDist.begin(), commutingDist.end()};
272  const auto cityIndexGen =
273  rn_manager->GetGenerator(locationDist, static_cast<unsigned>(omp_get_thread_num()));
274  location = cities[cityIndexGen()];
275  break;
276  }
277  case GridPoolType::Id::Workplace: {
278  std::vector<unsigned> commutingDist;
279  for (LocIndex destinationIndex = 0; destinationIndex < geogrid->NumLocations(); ++destinationIndex) {
280  commutingDist.emplace_back(geogrid->GetCommuteValue(locIndex, destinationIndex));
281  }
282  trng::fast_discrete_dist locationDist{commutingDist.begin(), commutingDist.end()};
283  const auto locationIndexGen =
284  rn_manager->GetGenerator(locationDist, static_cast<unsigned>(omp_get_thread_num()));
285  location = static_cast<LocIndex>(locationIndexGen());
286  break;
287  }
288  default: {
289  assert(false);
290  break;
291  }
292  }
293  // Pick a random pool from location.
294  const auto contactPoolRange = geogrid->GetContactPoolIdRange(poolType, location);
295  const trng::uniform_int_dist dist{static_cast<int>(contactPoolRange.first),
296  static_cast<int>(contactPoolRange.second)};
297  const auto poolIdGen = rn_manager->GetGenerator(dist, static_cast<unsigned>(omp_get_thread_num()));
298  return std::make_pair(location, static_cast<unsigned>(poolIdGen()));
299 }
