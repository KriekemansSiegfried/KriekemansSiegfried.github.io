37 shared_ptr<Population> PopBuilder::MakePoolSys(std::shared_ptr<Population> pop)
38 {
39  using namespace ContactPoolType;
40  auto& population = *pop;
41  auto& poolSys = population.GetContactPoolSys();
42 
43  // --------------------------------------------------------------
44  // Determine maximum pool ids in population.
45  // --------------------------------------------------------------
46  IdSubscriptArray<unsigned int> max_ids{0U};
47  for (const auto& p : population) {
48  for (Id typ : IdList) {
49  max_ids[typ] = max(max_ids[typ], p.GetPoolId(typ));
50  }
51  }
52  // --------------------------------------------------------------
53  // Initialize poolSys with empty ContactPools (even for Id=0).
54  // --------------------------------------------------------------
55  for (Id typ : IdList) {
56  for (size_t i = 0; i < max_ids[typ] + 1; i++) {
57  poolSys[typ].emplace_back(ContactPool(i, typ));
58  }
59  }
60 
61  // --------------------------------------------------------------
62  // Insert persons (pointers) in their contactpools. Having Id 0
63  // means "not belonging pool of that type" (e.g. school/ work -
64  // cannot belong to both, or e.g. out-of-work).
65  //
66  // Pools are uniquely identified by (typ, subscript) and a Person
67  // belongs, for typ, to pool with subscrip p.GetPoolId(typ).
68  // Defensive measure: we have a pool for Id 0 and leave it empty.
69  // --------------------------------------------------------------
70  for (auto& p : population) {
71  for (Id typ : IdList) {
72  const auto poolId = p.GetPoolId(typ);
73  if (poolId > 0) {
74  poolSys[typ][poolId].AddMember(&p);
75  }
76  }
77  }
78 
79  return pop;
80 }

82 shared_ptr<Population> PopBuilder::MakePersons(std::shared_ptr<Population> pop)
83 {
84  //------------------------------------------------
85  // Read persosns from file.
86  //------------------------------------------------
87  const auto belief_pt = m_config_pt.get_child("run.belief_policy");
88  const auto file_name = m_config_pt.get<string>("run.population_file");
89  const auto use_install_dirs = m_config_pt.get<bool>("run.use_install_dirs");
90  const auto file_path = (use_install_dirs) ? FileSys::GetDataDir() /= file_name : file_name;
91  if (!is_regular_file(file_path)) {
92  throw runtime_error(string(__func__) + "> Population file " + file_path.string() + " not present.");
93  }
94 
95  boost::filesystem::ifstream pop_file;
96  pop_file.open(file_path.string());
97  if (!pop_file.is_open()) {
98  throw runtime_error(string(__func__) + "> Error opening population file " + file_path.string());
99  }
100 
101  string line;
102  getline(pop_file, line); // step over file header
103  unsigned int person_id = 0U;
104 
105  while (getline(pop_file, line)) {
106  const auto values = Split(line, ",");
107  const auto age = FromString<unsigned int>(values[0]);
108  const auto household_id = FromString<unsigned int>(values[1]);
109  const auto school_id = FromString<unsigned int>(values[2]);
110  const auto work_id = FromString<unsigned int>(values[3]);
111  const auto primary_community_id = FromString<unsigned int>(values[4]);
112  const auto secondary_community_id = FromString<unsigned int>(values[5]);
113 
114  pop->CreatePerson(person_id, age, household_id, school_id, work_id, primary_community_id,
115  secondary_community_id);
116  ++person_id;
117  }
118 
119  pop_file.close();
120  return pop;
121 }

133  //------------------------------------------------

144  omp_set_schedule(omp_sched_static, 0);
145 
146  stride::util::RNManager::Info rn_info;

149 
150  const auto dataDir = stride::util::FileSys::GetDataDir();
151 
152  auto popgenerator = std::make_shared<stride::PopulationGenerator>(
153  popgenconfig, rn_manager, popgenconfig->get<unsigned>("popgen.population.size"), pop);
154 
155  popgenerator->GeneratePopulation();
156  population = popgenerator->GetPopulation();
157 
158  if (m_config_pt.get<std::string>("run.write_population_to_file", "") != "") {

162  auto grid = popgenerator->GetGrid();

164  }

166  grid->SetPersonIdLivesInLocation();
167  } else {

169  }
170  SurveySeeder(m_config_pt, m_rn_manager).Seed(MakePoolSys(population));
171  return pop;
