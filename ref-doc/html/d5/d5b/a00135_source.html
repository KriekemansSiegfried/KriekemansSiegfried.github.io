47 void DiseaseSeeder::Seed(std::shared_ptr<Population> pop)
48 {
49  // --------------------------------------------------------------
50  // Population immunity (natural immunity & vaccination).
51  // --------------------------------------------------------------
52  const auto immunityProfile = m_config_pt.get<std::string>("run.immunity_profile");
53  Vaccinate("immunity", immunityProfile, pop->GetContactPoolSys()[Id::Household]);
54 
55  const auto vaccinationProfile = m_config_pt.get<std::string>("run.vaccine_profile");
56  Vaccinate("vaccine", vaccinationProfile, pop->GetContactPoolSys()[Id::Household]);
57 
58  // --------------------------------------------------------------
59  // Seed infected persons.
60  // --------------------------------------------------------------
61  const auto sRate = m_config_pt.get<double>("run.seeding_rate");
62  const auto sAgeMin = m_config_pt.get<double>("run.seeding_age_min", 1);
63  const auto sAgeMax = m_config_pt.get<double>("run.seeding_age_max", 99);
64  const auto popSize = pop->size();
65  const auto maxPopIndex = static_cast<unsigned int>(popSize - 1);
66  auto generator = m_rn_manager.GetGenerator(trng::uniform_int_dist(0, maxPopIndex));
67  auto& logger = pop->GetContactLogger();
68 
69  auto numInfected = static_cast<unsigned int>(floor(static_cast<double>(popSize) * sRate));
70  while (numInfected > 0) {
71  Person& p = pop->at(static_cast<size_t>(generator()));
72  if (p.GetHealth().IsSusceptible() && (p.GetAge() >= sAgeMin) && (p.GetAge() <= sAgeMax)) {
73  p.GetHealth().StartInfection();

99  const auto popSize = pop->size();
100  const auto maxPopIndex = static_cast<unsigned int>(popSize - 1);
101  std::function<int()> generator;
102  auto numInfected = 0;
103  if (outbreak == 0) {
104  generator = m_rn_manager.GetGenerator(trng::uniform_int_dist(0, maxPopIndex));
105  numInfected = popSize;
106  } else if (outbreak == 5) {
107  generator = m_rn_manager.GetGenerator(trng::uniform_int_dist(
108  (trng::uniform_int_dist::result_type)multipop->GetIndices()[outbreak - 1], maxPopIndex));
109  numInfected = maxPopIndex - multipop->GetIndices()[outbreak - 1];
110  } else {
111  generator = m_rn_manager.GetGenerator(
112  trng::uniform_int_dist((trng::uniform_int_dist::result_type)multipop->GetIndices()[outbreak - 1],
113  (trng::uniform_int_dist::result_type)multipop->GetIndices()[outbreak]));
114  numInfected = multipop->GetIndices()[outbreak] - multipop->GetIndices()[outbreak - 1];
115  }

118  numInfected = static_cast<unsigned int>(floor(static_cast<double>(numInfected) * sRate));

130  std::vector<ContactPool>& immunityPools)
131 {
132  std::vector<double> immunityDistribution;
133  const double linkProbability = 0;
134  Immunizer immunizer(m_rn_manager);
135 
136  if (immunizationProfile == "Random") {
137  const auto immunityRate = m_config_pt.get<double>("run." + ToLower(immunityType) + "_rate");
138  for (unsigned int index_age = 0; index_age < 100; index_age++) {
139  immunityDistribution.push_back(immunityRate);
140  }
141  immunizer.Random(immunityPools, immunityDistribution, linkProbability);
142  } else if (immunizationProfile == "Cocoon") {
143  immunizer.Cocoon(immunityPools, immunityDistribution, linkProbability);
144  }
