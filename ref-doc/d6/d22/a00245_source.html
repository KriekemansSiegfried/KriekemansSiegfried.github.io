38 shared_ptr<Population> SurveySeeder::Seed(shared_ptr<Population> pop)
39 {
40  const string log_level = m_config_pt.get<string>("run.contact_log_level", "None");
41  if (log_level == "All" || log_level == "Susceptibles") {
42  Population& population = *pop;
43  auto& logger = population.GetContactLogger();
44  const auto max_index = static_cast<unsigned int>(population.size() - 1);
45  auto generator = m_rn_manager.GetGenerator(trng::uniform_int_dist(0, max_index));
46  const auto participants = m_config_pt.get<unsigned int>("run.num_participants_survey");
47 
48  // Use while-loop to get 'participants' unique participants (default sampling is with replacement).
49  // A for loop will not do because we might draw the same person twice.
50  auto num_samples = 0U;
51  while (num_samples < participants) {
52  Person& p = population[generator()];
53  if (!p.IsSurveyParticipant()) {
54  p.ParticipateInSurvey();
55  logger->info("[PART] {} {} {} {} {}", p.GetId(), p.GetAge(), p.GetGender(),
56  p.GetPoolId(Id::School), p.GetPoolId(Id::Work));
57  num_samples++;
58  }
59  }
60  }
61  return pop;
62 }
