86  matrix[commuting_order[i]][commuting_order[j]] =

92  for (auto& row : matrix) {

100 void stride::GeoGridBuilder::CalculateCommutingPopulations()
101 {
102  std::vector<unsigned int> commuting_out_absolute(m_grid->NumLocations());
103  std::vector<unsigned int> commuting_in_absolute(m_grid->NumLocations());
104 
105  for (LocIndex i = 0; i < m_grid->NumLocations(); ++i) {
106  /*
107  * first loop -> calculate denominator, commuting_out value
108  */
109  double denom{0.0};
110  for (LocIndex j = 0; j < m_grid->NumLocations(); ++j) {
111  denom += m_grid->GetCommuteValue(i, j);
112  }
113  double commute_out_fraction = 1.0 - static_cast<double>(m_grid->GetCommuteValue(i, i)) / denom;
114  commuting_out_absolute[i] = static_cast<unsigned int>(
115  std::round(commute_out_fraction * static_cast<double>(m_grid->GetLocationPopulationsize(i))));
116  /*
117  * second loop -> calculate absolute commuting_in values for all locations != i
118  */
119  for (LocIndex j = 0; j < m_grid->NumLocations(); ++j) {
120  if (i != j) {
121  double commuting_in_fraction =
122  static_cast<double>(m_grid->GetCommuteValue(i, j)) / denom;
123  commuting_in_absolute[j] += static_cast<unsigned>(std::round(
124  commuting_in_fraction * static_cast<double>(m_grid->GetLocationPopulationsize(i))));
125  }
126  }
127  }
128 
129  locationData<unsigned> commuting_size(m_grid->NumLocations());
130 
131  for (LocIndex i = 0; i < m_grid->NumLocations(); ++i) {
132  int current_pop =
133  m_grid->GetLocationPopulationsize(i) + commuting_in_absolute[i] - commuting_out_absolute[i];
134  commuting_size[i] = static_cast<unsigned>(current_pop);
135  if (current_pop < 1) {
136  std::cout << "Error" << std::endl;
137  }
138  }
139  m_grid->SetPopulationCommutingSize(commuting_size);
140 }

143 {

147  current_pop += m_grid->GetLocationPopulationsize(i);

149  const double multiplier = double(requested_population) / current_pop;

259 void stride::GeoGridBuilder::GenerateDistricts(unsigned int fracture_threshold, double L, double k, double mid)
260 {
261 
262  // function used for generating probability of being split up into districts
263  std::function<double(double)> logisticfunc = [L, k, mid](double x) {
264  return L / (1 + std::exp(-1 * k * (x - mid)));
265  };
266 
267  // ID for first new location, incremented for each following new location
268  auto newindex = m_grid->GetLocationID(m_grid->NumLocations() - 1) + 1;
269 
270  // iterate over all locationsu
271  unsigned int numberOfOriginals = m_grid->NumLocations();
272 
273  for (unsigned int index = 0; index < numberOfOriginals; index++) {
274  const auto loc_pop = m_grid->GetLocationPopulationsize(index);
275  const auto loc_name = m_grid->GetLocationName(index);
276  if (GetsFractured(loc_pop, fracture_threshold, logisticfunc)) {
277  const auto nrOfDistrictsDist = NumberDistrictsDist(loc_pop, fracture_threshold);
278  const auto generator = rn_manager->GetGenerator(nrOfDistrictsDist);
279  const auto numberOfDistricts = generator() + 1;
280  if (numberOfDistricts <= 1) {
281  // no splitting required
282  continue;
283  }
284  const auto districtpopulations =
285  GetDistrictPopulations(static_cast<unsigned int>(numberOfDistricts), loc_pop);
286  // for each new population, add district
287  unsigned int namecounter = 1;
288  for (const auto newpop : districtpopulations) {
289  std::string newname = loc_name + std::to_string(namecounter++);
290  AddDistrict(index, newindex, newname, newpop);
291  newindex++;
292  }
293 
294  // delete old location
295  m_grid->DeleteLocation(index);
296  numberOfOriginals--;
297  }
298  }
299 }

301 trng::fast_discrete_dist stride::GeoGridBuilder::NumberDistrictsDist(const unsigned int pop, const unsigned int minfrac)
302 {
304  std::vector<double> distribution;
305  auto ratio = static_cast<double>(pop) / static_cast<double>(minfrac);
306 
307  double mu = ratio / 3.0;
308  double sigma = ratio / 3.0;
309  trng::normal_dist<double> normalDist(mu, sigma);
310  auto fracAmount = static_cast<unsigned int>(std::floor(pop / minfrac));
311  for (unsigned int i = 0; i < fracAmount; i++) {
312  distribution.push_back(normalDist.pdf(i));
313  }
314  return trng::fast_discrete_dist(distribution.begin(), distribution.end());
315 }

317 std::vector<unsigned int> stride::GeoGridBuilder::GetDistrictPopulations(const unsigned int& amountOfFracs,
318  const unsigned long& pop)
319 {
320  double mu = pop / static_cast<double>(amountOfFracs) / 2.0;
321  double stdv = (pop / static_cast<double>(amountOfFracs)) / 4.0;
322  trng::normal_dist<double> normalDist(mu, stdv);
323  auto generator = rn_manager->GetGenerator(normalDist);
324  std::vector<unsigned int> pops;
325  unsigned long total = 0;
326  for (unsigned int i = 0; i < amountOfFracs; i++) {
327  auto subtract = generator();
328  if (subtract > pop / static_cast<double>(amountOfFracs)) {
329  subtract =
330  pop / static_cast<double>(amountOfFracs) / 5;
331  }
332  auto currentPop = static_cast<unsigned int>(pop / static_cast<double>(amountOfFracs) - subtract);
333  pops.push_back(static_cast<unsigned int&&>(currentPop));
334  total += currentPop;
335  }
336  for (auto& newPop : pops) {
337  newPop = static_cast<unsigned int>(newPop * pop / total);
338  }
339  return pops;
340 }
